main:
    j __user_main    # Jump to main function
g:
    # Function prologue
    addi $sp, $sp, -8    # allocate stack frame
    sw $ra, 4($sp)    # save return address
    sw $fp, 0($sp)    # save frame pointer
    addi $fp, $sp, 8    # set new frame pointer

    # Allocate n_7
    # Allocate n_7

    # Temp_0 := n_7
    lw $t0, n_7

    # p_10 := Temp_0
    sw $t0, p_10

    # Temp_1 := PrintInt()
    # Save caller-saved registers
    addi $sp, $sp, -40    # allocate space for $t0-$t9

    sw $t4, 16($sp)

    sw $t8, 32($sp)
    sw $t9, 36($sp)

    # Restore caller-saved registers
    lw $t1, 4($sp)
    lw $t2, 8($sp)
    lw $t3, 12($sp)
    lw $t4, 16($sp)
    lw $t5, 20($sp)
    lw $t6, 24($sp)
    lw $t7, 28($sp)
    lw $t8, 32($sp)
    lw $t9, 36($sp)
    addi $sp, $sp, 40    # deallocate register save area

    lw $t1, n_7

    li $t0, 2

    # Temp_2 := Temp_3 == Temp_4
    move $t9, $t0

    # Function epilogue
    lw $ra, 4($sp)    # restore return address
    lw $fp, 0($sp)    # restore frame pointer

    jr $ra

    # Temp_6 := n_7
    lw $t0, n_7

    li $t0, 32767
    j saturate_sub_done_4
saturate_sub_min_3:
    li $t0, -32768
saturate_sub_done_4:

    # n_7 := Temp_5
    # Save n_7 before store (recursive call ahead)
    addi $sp, $sp, -4    # allocate space
    lw $t8, n_7    # load current value
    sw $t8, 0($sp)    # save on stack
    sw $t0, n_7

    # Temp_8 := g()
    # Save caller-saved registers
    addi $sp, $sp, -40    # allocate space for $t0-$t9
    sw $t0, 0($sp)
    sw $t1, 4($sp)

    sw $t3, 12($sp)

    move $t0, $v0    # save return value

    lw $t1, 4($sp)
    lw $t2, 8($sp)
    lw $t3, 12($sp)
    lw $t4, 16($sp)
    lw $t5, 20($sp)
    lw $t6, 24($sp)
    lw $t7, 28($sp)
    lw $t8, 32($sp)
    lw $t9, 36($sp)
    addi $sp, $sp, 40    # deallocate register save area
    # Restore n_7 after call (from pre-store save)
    lw $t8, 0($sp)    # load saved value
    sw $t8, n_7    # restore global
    addi $sp, $sp, 4    # deallocate global save space
